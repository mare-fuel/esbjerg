In [14]:
# load maritime cargo data from eurostat to pandas
tab = 'mar_go_am_'
url = 'https://ec.europa.eu/eurostat/estat-navtree-portlet-prod/BulkDownloadListing?file=data/'
ext = '.tsv.gz'
cd = pd.DataFrame()

for c in countries:
    file = tab + c + ext
    localpath = os.path.join('input', file)
    targeturl = os.path.join(url, file)
    if not os.path.exists(localpath):
        print('Downloading cargo data from Eurostat for ' + c)
        urllib.request.urlretrieve(targeturl, localpath)
    else:
        print('Loading local cargo data file for ' + c)
    tmp = pd.read_csv(localpath, sep='\t|\s\t|,', compression='gzip', engine='python', encoding='utf8')
    tmp['geo'] = c
    cd = cd.append(tmp, sort=False)
print('Done')
